## This script reads in a file containing HSP runs and for each staffing 
## level and due-date range outputs a file containing the non-zero
## PLB(\thsp) values so that a histogram can be computed.

import itertools
import numpy
import os
import string
import sys
##

def extract_VFBData(indVals):
    '''This function extracts the relevant VFB run data from the input line 
    which has already been split into a indVals a vector of individual values.'''

    oldVFBCode= True
    isOpt= True
    lbLmax, vfbLmax, tStarLmax= 0, 199999, 0
    vfbTime= 0

    ## Determine if the runs use the old VFB code where \lblmax was being
    ## stored improperly or the new VFB code where it is stored properly
    for value in indVals:
        if value == 'OptAllocFound':
            break

        if value == 'TStarIsOpt':
            oldVFBCode= False
            break

    if oldVFBCode:
        # Extract the necessary values from the line
        for i, val in enumerate(indVals):
            # Takes care of the case where allocation \tstar was optimal
            if val == 'OptAllocFound':
                if indVals[i+1] == '0':
                    isOpt= False

            if val == 'LB(Lmax)':  ## This finds VFB(Lmax)
                vfbLmax= int(indVals[i+3])

            if val == 'MinLBAlloc':  ## This finds LB(Lmax)
                lbLmax= int(indVals[i+1])
                tStarLmax= int(indVals[i+2])

            if val == 'EnumerationRunTime':
                vfbTime= int(indVals[i+1])
                break
    else:
        # Extract the necessary values from the line
        for i, val in enumerate(indVals):
            # Takes care of the case where allocation \tstar was optimal
            if val == 'TStarIsOpt':
                if indVals[i+1] == '0':
                    isOpt= False

            if val == 'LB(Lmax)':
                lbLmax= int(indVals[i+1])     ## LB(Lmax)
                vfbLmax= int(indVals[i+3])    ## VF(\tvfb)

            if val == 'AllocTStar':             
                tStarLmax= int(indVals[i+2])  ## VF(\tstar)

            if val == 'EnumRunTime':
                vfbTime= int(indVals[i+1])
                break

    return isOpt, lbLmax, tStarLmax, vfbLmax, vfbTime


def extract_HSP2Data(indVals):
    '''This function extracts the relevant HSP2 run data from the input line 
    which has already been split into a vector of individual values (indVals).
    NOTE: The HSP2 data must have been generated by vf_workerAlloc_v2.0'''

    isOpt= True
    lbLmax, sStratLmax, hsp2Lmax= 0, 199999, 0

    # Extract the necessary values from the line
    for i, val in enumerate(indVals):
        # Takes care of the case where allocation \tstar was optimal
        if val == 'TStarIsOpt':
            if indVals[i+1] == '1':
                lbLmax= int(indVals[i+10])
                # hsp2Lmax is the minimum of the Lmax value from the initial
                # allocation and allocation \tstar
                hsp2Lmax= min(int(indVals[i+6]), int(indVals[i+11]))                    
                break

        if val == 'AllocTStar':
            isOpt= False
            lbLmax= int(indVals[i+1])


        if val in ['LNSS_Lmax', 'QSS1_Lmax', 'QSS2_Lmax']:
            # Iterate over the Lmax values checking to see if any is
            # strictly smaller than the current value of sStratLmax
            for q in range(2, int(indVals[i+1])+2):
                if (int(indVals[i+q]) < sStratLmax):
                    hsp2Lmax= int(indVals[i+q])

        if val == 'ThetaHSP_Lmax':
            hsp2Lmax= int(indVals[i+1])

            # Data analysis check to make sure vf_workerAlloc_v2.0 code 
            # is working.
            if sStratLmax < hsp2Lmax:
                print "vf_workerAlloc_v2.0 code incorrect"

            break

    return isOpt, lbLmax, hsp2Lmax

def main():
    # Generate the input and file names
    vfbFileNames, hsp2FileNames= [], []
    for m in ('1a',):
        for j in ('asym', 'sym'):
            for k in ('60', '70', '80', '90'):      
                vfbFileNames.append('vfb_' + m + '_' + j + '_' + k + 'pct_500reps.txt')
                hsp2FileNames.append('hsp2_' + m + '_' + j + '_' + k + 'pct_500reps.txt')       

    dataFile= open('jss_data.csv', 'w')
    dataFile.write('jobs,machs,grps,wrkrs,ddr,rep,bal,tstar_is_opt,lb_lmax,tstar_lmax,vfb_lmax,hsp2_lmax\n')

    for vfbFileNm, hsp2FileNm in itertools.izip(vfbFileNames, hsp2FileNames):
        vfbFile= open('./data/' + vfbFileNm, 'r')
        hsp2File= open('./data/' + hsp2FileNm, 'r')

        print vfbFileNm, hsp2FileNm

        for vfbLine, hsp2Line in itertools.izip(vfbFile, hsp2File):
            vfbIndVals= vfbLine.split()
            hsp2IndVals= hsp2Line.split()

            if int(vfbIndVals[15]) < 10001:
                for idx in [0, 1, 2, 3, 8, 15]:
                    dataFile.write(str(vfbIndVals[idx]) + ',')
                if int(vfbIndVals[11]) == 0:
                    dataFile.write('SYM,')
                else:
                    dataFile.write('ASYM,')

                vfbLineData= extract_VFBData(vfbIndVals)
                if vfbLineData[0]:
                    dataFile.write('T,')
                else:
                    dataFile.write('F,')
                dataFile.write(str(vfbLineData[1]) + ',')
                dataFile.write(str(vfbLineData[2]) + ',')
                dataFile.write(str(vfbLineData[3]) + ',')

                hsp2LineData= extract_HSP2Data(hsp2IndVals)
                if vfbLineData[0]:
                    dataFile.write('\n')
                else:
                    dataFile.write(str(hsp2LineData[2]) + '\n')

    dataFile.close()

if __name__ == "__main__":
    main()
